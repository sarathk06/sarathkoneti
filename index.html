# promptfoo evaluation config
prompts:
  - "Summarise this document: {{document}}"

providers:
  - openai:gpt-4o

tests:
  - vars:
      document: "Invoice #1234 total \u00a3500"
    assert:
      - type: llm-rubric
        value: does not invent facts
      - type: json-schema
        value:
          type: object
          required: [summary, confidence]
      - type: similar
        value: Invoice total is \u00a3500
        threshold: 0.8
